This function checks if any letters than indicate singular words are in the words and then transforms them to plural

import pandas as pd

def pluralize_gift(gift):
    """
    Returns plural of a noun

    Parameters
    ----------
    gift: str
        A noun

    Return
    ------
    str
        Plural version
    """

    if isinstance(gift, str):
        obj_1 = gift
        if obj_1.find("oo") != -1:
            return obj_1.replace("oo", "ee")
        elif obj_1.endswith("y"):
            return obj_1[:-1] + "ies"
        else:
            return obj_1 + "s"

#Using this to process vectorized data
    elif isinstance(gift, pd.Series):
        pluralized = gift + "s"
        y_mask = gift.str.endswith('y')
        pluralized[y_mask] = gift[y_mask].str[:-1] + "ies"
        oo_mask = gift.str.contains("oo")
        pluralized[oo_mask] = gift[oo_mask].str.replace("oo", "ee")
        return pluralized

    return gift

def make_phrase(day, num_word, item, verb, adjective, location):
    """
    Returns a phrase with a gift phrase

    Parameters
    ----------
    day: integer 1-12
    other: strings

    Return
    ------
    str
        A phrase

    """
    VOWELS = 'aeiouAEIOU'

#Used chat to troublshoot and fix code here because it was giving me a lot of trouble
    if isinstance(day, pd.Series):
        verb = verb.fillna("");
        adjective = adjective.fillna("");
        location = location.fillna("")
    else:
        verb = verb if verb else "";
        adjective = adjective if adjective else "";
        location = location if location else ""

    if isinstance(day, pd.Series):
        plural_mask = (day > 1); pluralized_item = item.copy()
        pluralized_item[plural_mask] = pluralize_gift(item[plural_mask]); item = pluralized_item
    elif day > 1:
        item = pluralize_gift(item)

    if isinstance(day, pd.Series):
        starts_with_vowel = item.str[0].isin(list(VOWELS))
        prefix = pd.Series(num_word, index=day.index); day_1_mask = (day == 1)
        prefix[day_1_mask & starts_with_vowel] = "an"; prefix[day_1_mask & (~starts_with_vowel)] = "a"
    else:
        starts_with_vowel = item[0] in VOWELS
        prefix = "an" if day == 1 and starts_with_vowel else ("a" if day == 1 else num_word)

    if isinstance(day, pd.Series):
        parts_df = pd.DataFrame({'prefix': prefix, 'adjective': adjective, 'item': item, 'verb': verb, 'location': location})
        return parts_df.apply(
            lambda row: " ".join(str(x).strip() for x in row if str(x).strip()),
            axis=1
        ).str.strip()
    else:
        parts = [prefix, adjective, item, verb, location]
        return " ".join(part for part in parts if part).strip()

O_MAP = {
    1: "first", 2: "second", 3: "third", 4: "fourth", 5: "fifth",
    6: "sixth", 7: "seventh", 8: "eighth", 9: "ninth", 10: "tenth",
    11: "eleventh", 12: "twelfth"
}
MAP = {
    'first': 'a', 'second': 'two', 'third': 'three', 'fourth': 'four',
    'fifth': 'five', 'sixth': 'six', 'seventh': 'seven', 'eighth': 'eight',
    'ninth': 'nine', 'tenth': 'ten', 'eleventh': 'eleven', 'twelfth': 'twelve'
}

def sing_day(dataset, num, phrase_col):


    num_word = O_MAP.get(num, str(num) + "th")
    intro = "On the " + num_word + " day of Christmas, my true love sent to me"

    gifts_data = dataset.loc[dataset['Day'] <= num].sort_values(by='Day', ascending=False)
    gifts_list = gifts_data[phrase_col].tolist()

    if not gifts_list: return intro.strip()

    gifts_stanza = ""

    total_gifts = len(gifts_list)

    for i in range(total_gifts):
        phrase = gifts_list[i]

        if i < total_gifts - 2:
            gifts_stanza += phrase

        elif i == total_gifts - 2:
            gifts_stanza += phrase

        else:
            gifts_stanza += phrase


    if num == 1:
        gifts_stanza = gifts_list[0]
        return intro + gifts_stanza

    return intro + gifts_stanza
